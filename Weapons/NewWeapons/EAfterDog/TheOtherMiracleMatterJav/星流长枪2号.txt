        public override void AI()
        {
            Projectile.rotation = Projectile.velocity.ToRotation() + MathHelper.PiOver4;

            if (stage == 0) // 第1阶段：快速减速
            {
                Projectile.velocity *= 0.97f;
                stageTimer++;
                if (stageTimer >= 15)
                {
                    stage = 1;
                    initialVelocity = Projectile.velocity / 0.97f / 15f; // 记录初始速度
                    Projectile.velocity = initialVelocity * 200f; // 进入第2阶段，速度加倍
                    stageTimer = 0;

                    // 生成4个尖刺型粒子特效
                    for (int i = 0; i < 4; i++)
                    {
                        Vector2 velocity = new Vector2(0, -5).RotatedBy(MathHelper.ToRadians(90 * i));
                        PointParticle particle = new PointParticle(Projectile.Center, velocity * 0.5f, false, 5, 1.1f, Color.White);
                        GeneralParticleHandler.SpawnParticle(particle);
                    }
                }
            }
            else if (stage == 1) // 第2阶段：冲刺并反弹
            {
                if (CheckScreenBounds())
                {
                    bounces++;
                    if (bounces <= 3)
                    {
                        Vector2 direction = Vector2.Reflect(Projectile.velocity, GetCollisionNormal());
                        Projectile.velocity = direction;

                        // 反弹时发射ExoFlareCluster
                        Projectile.NewProjectile(Projectile.GetSource_FromThis(), Projectile.Center, Projectile.velocity, ModContent.ProjectileType<ExoFlareCluster>(), Projectile.damage, Projectile.knockBack, Projectile.owner);
                    }
                    else
                    {
                        LockOnToNearestEnemy(); // 第3次反弹后锁定敌人
                    }
                }
            }
        }

        private bool CheckScreenBounds()
        {
            Vector2 screenBoundsMin = Main.screenPosition;
            Vector2 screenBoundsMax = Main.screenPosition + new Vector2(Main.screenWidth, Main.screenHeight);

            if (Projectile.position.X <= screenBoundsMin.X || Projectile.position.X + Projectile.width >= screenBoundsMax.X ||
                Projectile.position.Y <= screenBoundsMin.Y || Projectile.position.Y + Projectile.height >= screenBoundsMax.Y)
            {
                return true;
            }
            return false;
        }

        private Vector2 GetCollisionNormal()
        {
            // 返回碰撞的法线，用于计算反弹方向
            return new Vector2(-Projectile.velocity.X, -Projectile.velocity.Y).SafeNormalize(Vector2.UnitX);
        }

        private void LockOnToNearestEnemy()
        {
            NPC target = Projectile.Center.ClosestNPCAt(1800);
            if (target != null)
            {
                Vector2 direction = (target.Center - Projectile.Center).SafeNormalize(Vector2.Zero);
                Projectile.velocity = Vector2.Lerp(Projectile.velocity, direction * 12f, 0.08f);
            }
        }

        public override void OnHitNPC(NPC target, NPC.HitInfo hit, int damageDone)
        {
            bool overlapWithTile = Collision.SolidCollision(target.position, target.width, target.height);
            if (overlapWithTile)
            {
                // 造成1.5倍伤害
                damageDone = (int)(damageDone * 1.5f);

                // 在敌人头顶释放ExoFire
                for (int i = 0; i < 3; i++)
                {
                    Vector2 offset = new Vector2(Main.rand.Next(-20, 20), -50);
                    Projectile.NewProjectile(Projectile.GetSource_FromThis(), target.Center + offset, Vector2.UnitY * 5f, ModContent.ProjectileType<ExoFire>(), (int)(Projectile.damage * 1.0f), Projectile.knockBack, Projectile.owner);
                }
            }
        }
