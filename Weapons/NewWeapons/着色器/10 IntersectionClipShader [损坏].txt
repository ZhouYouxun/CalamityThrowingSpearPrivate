   //  π”√ Shader: IntersectionClipShader ("CalamityMod:IntersectionClip")
        public override bool PreDraw(ref Color lightColor)
        {
            GameShaders.Misc["CalamityMod:IntersectionClip"].SetShaderTexture(
                ModContent.Request<Texture2D>("CalamityMod/ExtraTextures/Trails/FabstaffStreak")
            );

            Vector2 overallOffset = Projectile.Size * 0.5f + Projectile.velocity * 1.2f;
            int numPoints = 35;

            PrimitiveRenderer.RenderTrail(
                Projectile.oldPos,
                new(
                    IntersectionWidthFunction,
                    IntersectionColorFunction,
                    (completionRatio, vertexPos) => overallOffset,
                    shader: GameShaders.Misc["CalamityMod:IntersectionClip"]
                ),
                numPoints
            );

            return false;
        }

        private float IntersectionWidthFunction(float completionRatio)
        {
            return 20f * (1f - completionRatio);
        }

        private Color IntersectionColorFunction(float completionRatio)
        {
            return Color.Lerp(Color.Cyan, Color.Transparent, completionRatio);
        }
