   //  π”√ Shader: LocalLinearTransformationShader ("CalamityMod:LinearTransformation")
        public override bool PreDraw(ref Color lightColor)
        {
            GameShaders.Misc["CalamityMod:LinearTransformation"].SetShaderTexture(
                ModContent.Request<Texture2D>("CalamityMod/ExtraTextures/Trails/FabstaffStreak")
            );

            Vector2 overallOffset = Projectile.Size * 0.5f + Projectile.velocity * 1.3f;
            int numPoints = 38;

            PrimitiveRenderer.RenderTrail(
                Projectile.oldPos,
                new(
                    LinearWidthFunction,
                    LinearColorFunction,
                    (completionRatio, vertexPos) => overallOffset,
                    shader: GameShaders.Misc["CalamityMod:LinearTransformation"]
                ),
                numPoints
            );

            return false;
        }

        private float LinearWidthFunction(float completionRatio)
        {
            return 18f * (1f - completionRatio * 0.5f);
        }

        private Color LinearColorFunction(float completionRatio)
        {
            return Color.Lerp(Color.Green, Color.Transparent, completionRatio);
        }
