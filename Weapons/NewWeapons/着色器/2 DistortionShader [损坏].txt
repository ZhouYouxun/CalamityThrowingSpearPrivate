        public override bool PreDraw(ref Color lightColor)
        {
            // °у¶Ё `DistortionShader`
            GameShaders.Misc["CalamityMod:DistortionShader"].SetShaderTexture(
                ModContent.Request<Texture2D>("CalamityMod/ExtraTextures/Trails/FabstaffStreak")
            );

            Vector2 overallOffset = Projectile.Size * 0.5f + Projectile.velocity * 1.5f;
            int numPoints = 40; // НПОІіӨ¶ИёьіӨТФККЕдЕӨЗъР§№ы

            PrimitiveRenderer.RenderTrail(
                Projectile.oldPos,
                new(
                    DistortionWidthFunction,  // јЖЛгНПОІҝн¶И
                    DistortionColorFunction,  // јЖЛгНПОІСХЙ«
                    (_) => overallOffset,
                    shader: GameShaders.Misc["CalamityMod:DistortionShader"] // °у¶Ё `DistortionShader`
                ),
                numPoints
            );

            return false;
        }


        private float DistortionWidthFunction(float completionRatio)
        {
            float baseWidth = 20f;
            float pulseEffect = (float)Math.Sin(completionRatio * 8f + Main.GlobalTimeWrappedHourly * 2f) * 4f;
            return baseWidth + pulseEffect; // ИГҝн¶ИВц¶Ҝұд»Ҝ
        }

        private Color DistortionColorFunction(float completionRatio)
        {
            float distortFactor = (float)Math.Cos(completionRatio * 5f - Main.GlobalTimeWrappedHourly * 3f) * 0.5f + 0.5f;
            Color baseColor = Color.Lerp(Color.Purple, Color.BlueViolet, distortFactor);
            return Color.Lerp(baseColor, Color.Transparent, completionRatio);
        }
