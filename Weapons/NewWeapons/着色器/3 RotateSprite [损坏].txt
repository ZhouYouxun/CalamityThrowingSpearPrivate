        public override bool PreDraw(ref Color lightColor)
        {
            // АѓЖЈ `RotateSprite`
            GameShaders.Misc["CalamityMod:RotationPass"].SetShaderTexture(
                ModContent.Request<Texture2D>("CalamityMod/ExtraTextures/Trails/FabstaffStreak")
            );

            Vector2 overallOffset = Projectile.Size * 0.5f + Projectile.velocity * 1.1f;
            int numPoints = 35; // ЭЯЮВТдЖЬвдЪЪХфа§зЊаЇЙћ

            PrimitiveRenderer.RenderTrail(
                Projectile.oldPos,
                new(
                    RotateWidthFunction,  // МЦЫуЭЯЮВПэЖШ
                    RotateColorFunction,  // МЦЫуЭЯЮВбеЩЋ
                    (_) => overallOffset,
                    shader: GameShaders.Misc["CalamityMod:RotationPass"] // АѓЖЈ `RotateSprite`
                ),
                numPoints
            );

            return false;
        }

        private float RotateWidthFunction(float completionRatio)
        {
            float baseWidth = 22f;
            float rotationEffect = (float)Math.Cos(completionRatio * 6f + Main.GlobalTimeWrappedHourly * 2f) * 3f;
            return baseWidth + rotationEffect; // ШУЭЯЮВПэЖШЫца§зЊаЇЙћВЈЖЏ
        }
        private Color RotateColorFunction(float completionRatio)
        {
            float rotationIntensity = (float)Math.Sin(completionRatio * 7f + Main.GlobalTimeWrappedHourly * 3f) * 0.5f + 0.5f;
            Color baseColor = Color.Lerp(Color.Orange, Color.Red, rotationIntensity);
            return Color.Lerp(baseColor, Color.Transparent, completionRatio);
        }
