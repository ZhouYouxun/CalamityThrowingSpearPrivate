    // К№УГ Shader: FadingSolidTrailShader ("CalamityMod:FadingSolidTrail")
        public override bool PreDraw(ref Color lightColor)
        {
            GameShaders.Misc["CalamityMod:FadingSolidTrail"].SetShaderTexture(
                ModContent.Request<Texture2D>("CalamityMod/ExtraTextures/Trails/FabstaffStreak")
            );

            Vector2 overallOffset = Projectile.Size * 0.5f + Projectile.velocity * 1.1f;
            int numPoints = 32; // ҪҘТюНПОІККЕдҪП¶М№мјЈ

            PrimitiveRenderer.RenderTrail(
                Projectile.oldPos,
                new(
                    FadingWidthFunction,
                    FadingColorFunction,
                    (completionRatio, vertexPos) => overallOffset,
                    shader: GameShaders.Misc["CalamityMod:FadingSolidTrail"]
                ),
                numPoints
            );

            return false;
        }

        private float FadingWidthFunction(float completionRatio)
        {
            float baseWidth = 16f;
            return baseWidth * (1f - completionRatio); // ОІ¶ЛЦрҪҘұдПё
        }

        private Color FadingColorFunction(float completionRatio)
        {
            Color baseColor = Color.LightGray;
            return Color.Lerp(baseColor, Color.Transparent, completionRatio); // СХЙ«ЦрҪҘНёГч
        }
