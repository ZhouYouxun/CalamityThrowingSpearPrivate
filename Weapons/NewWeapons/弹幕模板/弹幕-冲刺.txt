using Microsoft.Xna.Framework;
using Terraria;
using Terraria.ID;
using Terraria.ModLoader;

namespace CalamityThrowingSpear.Projectiles.Tutorial
{
    public class TutorialDashProjectile : ModProjectile
    {
        private const int MaxChargeTime = 60; // 蓄力时长
        private const int MaxDashTime = 60;   // 冲刺时长
        private const float DashSpeed = 25f;  // 冲刺速度

        private int dashTime = 0;
        private Vector2 lockedDirection;

        public override void SetStaticDefaults()
        {
            ProjectileID.Sets.TrailCacheLength[Projectile.type] = 10;
            ProjectileID.Sets.TrailingMode[Projectile.type] = 1;
        }

        public override void SetDefaults()
        {
            Projectile.width = Projectile.height = 32;
            Projectile.friendly = true;
            Projectile.penetrate = -1;
            Projectile.timeLeft = MaxChargeTime + MaxDashTime;
            Projectile.tileCollide = false;
            Projectile.ignoreWater = true;
        }

        public override void AI()
        {
            Player owner = Main.player[Projectile.owner];

            if (Projectile.velocity == Vector2.Zero) // 蓄力阶段
            {
                Projectile.rotation = Projectile.velocity.ToRotation() + MathHelper.PiOver4; // 45° 对齐
                // 或使用垂直对齐：Projectile.rotation = Projectile.velocity.ToRotation() + MathHelper.PiOver2 + MathHelper.Pi;

                Projectile.Center = owner.MountedCenter;
                owner.heldProj = Projectile.whoAmI;

                if (Projectile.ai[0]++ >= MaxChargeTime || !owner.channel)
                {
                    StartDash(owner);
                }
            }
            else // 冲刺阶段
            {
                Projectile.tileCollide = true;
                Projectile.friendly = true;

                Projectile.velocity = lockedDirection * DashSpeed;
                owner.velocity = Projectile.velocity;
                owner.Center = Projectile.Center;

                dashTime++;
                if (dashTime >= MaxDashTime)
                {
                    Projectile.Kill();
                }
            }
        }

        private void StartDash(Player owner)
        {
            lockedDirection = (Main.MouseWorld - Projectile.Center).SafeNormalize(Vector2.UnitY);
            Projectile.velocity = lockedDirection * DashSpeed;
            dashTime = 0;
            Projectile.netUpdate = true;
        }

        public override void OnHitNPC(NPC target, NPC.HitInfo hit, int damageDone)
        {
            // 命中敌人时触发，可用于附加 Buff 或播放特效
        }

        public override void OnKill(int timeLeft)
        {
            // 弹幕结束时触发，可用于生成粒子或音效
        }

        public override bool PreDraw(ref Color lightColor)
        {
            // 控制绘制行为（如拖尾、自定义贴图），默认可直接返回 true
            return true;
        }
    }
}
